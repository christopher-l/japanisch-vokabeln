\newcommand{\vocab@}[3]{

  {\large#1}\\%[-.4em]
  \raggedright
  \setcounter{translations}{0}

}

\newcommand{\translation}[1]{

  \vspace{-1em}
  \begin{enumerate}[leftmargin=*]
    \setcounter{enum\romannumeral\the\@enumdepth}{\value{translations}}
    \stepcounter{translations}
    \itemsep0em \item #1
  \end{enumerate}
  \vspace{-1em}
  \else
  #1
  \fi}
\makeatother
